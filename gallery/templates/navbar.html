class Article(models.Model): title = models.CharField(max_length = 60) post = models.TextField() editor = models.ForeignKey(Editor, on_delete = models.CASCADE) tags = models.ManyToManyField(tags) pub_date = models.DateTimeField(auto_now_add = True) article_image = models.ImageField(upload_to = 'articles/',default = 'image' )
